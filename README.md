## Multithreading
```py
from dankware import multithread
import time

a = 0
def example():
    global a
    a += 1
    print(a)
    time.sleep(5)
        
multithread(example, 10)
```

```py
from dankware import multithread
import time

list = [1, 2, 3, 4, 5];sum = 0

def example(num):
    global sum;sum += num;time.sleep(5)

multithread(example, 10, list)
print(sum)
```
